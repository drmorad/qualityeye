function ripple(e){
  const el=e.currentTarget,r=el.getBoundingClientRect();
  const cx=(e.touches?e.touches[0].clientX:e.clientX)-r.left;
  const cy=(e.touches?e.touches[0].clientY:e.clientY)-r.top;
  const sz=Math.max(r.width,r.height)*2.2;
  const sp=document.createElement('span');
  sp.className='rpl-el';
  sp.style.cssText=`width:${sz}px;height:${sz}px;left:${cx-sz/2}px;top:${cy-sz/2}px`;
  el.appendChild(sp);setTimeout(()=>sp.remove(),650);
}

}

function handleDocSelect(e){
  const file=e.target.files&&e.target.files[0];
  if(!file) return;
  if(!file.type.includes('pdf')&&!file.name.toLowerCase().endsWith('.pdf')){
    toast(lang==='ar'?'يرجى اختيار ملف PDF فقط':'Please select a PDF file','bad'); return;
  }
  if(!apiKey||isDemoMode){
    toast(lang==='ar'?'يلزم مفتاح API حقيقي لرفع الوثائق':'Real API key required to upload documents','bad'); return;
  }
  uploadDocFile(file);
  e.target.value='';
}

async function uploadDocFile(file){
  const pill=document.getElementById('uploadingPill');
  pill.classList.add('on');
  try{
    const formData=new FormData();
    formData.append('file',file,file.name);

    const resp=await fetch('https://api.anthropic.com/v1/beta/files',{
      method:'POST',
      headers:{
        'x-api-key':apiKey,
        'anthropic-version':'2023-06-01',
        'anthropic-beta':FILES_BETA,
        'anthropic-dangerous-direct-browser-access':'true'
      },
      body:formData
    });

    if(!resp.ok){
      const err=await resp.json().catch(()=>({}));
      throw new Error(err.error?.message||'HTTP '+resp.status);
    }

    const data=await resp.json();
    const doc={
      fileId:data.id,
      name:file.name,
      size:file.size,
      uploadedAt:new Date().toLocaleString(lang==='ar'?'ar-SA':'en-GB')
    };
    complianceDocs.push(doc);
    await saveDocsToStorage();
    updateDocsUI();
    toast(lang==='ar'?`✓ تم رفع "${file.name}" بنجاح`:`✓ "${file.name}" uploaded successfully`,'doc');
  } catch(err){
    toast((lang==='ar'?'فشل رفع الوثيقة: ':'Upload failed: ')+err.message,'bad');
  } finally{
    pill.classList.remove('on');
  }
}

async function deleteDoc(fileId){
  // Delete from Anthropic Files API
  if(apiKey&&!isDemoMode){
    try{
      await fetch(`https://api.anthropic.com/v1/beta/files/${fileId}`,{
        method:'DELETE',
        headers:{
          'x-api-key':apiKey,
          'anthropic-version':'2023-06-01',
          'anthropic-beta':FILES_BETA,
          'anthropic-dangerous-direct-browser-access':'true'
        }
      });
    } catch(e){}
  }
  complianceDocs=complianceDocs.filter(d=>d.fileId!==fileId);
  await saveDocsToStorage();
  updateDocsUI();
  toast(lang==='ar'?'تم حذف الوثيقة':'Document deleted','inf');
}

function updateDocsUI(){
  const cnt=complianceDocs.length;
  // Topbar badge
  const badge=document.getElementById('docBadgePill');
  badge.textContent=`📎 ${cnt}`;
  badge.classList.toggle('on',cnt>0);
  // Area bar indicator
  const ind=document.getElementById('docsIndicator');
  const indLbl=document.getElementById('docsIndicatorLbl');
  ind.classList.toggle('on',cnt>0);
  if(cnt>0) indLbl.textContent=`${cnt} ${t('docsActive').toUpperCase()}`;
  // Camera HUD
  const hud=document.getElementById('camDocsHud');
  const hudCnt=document.getElementById('camDocsCount');
  hud.classList.toggle('on',cnt>0);
  if(hudCnt) hudCnt.textContent=cnt;
  // Doc list in settings
  renderDocList();
}

function renderDocList(){
  const list=document.getElementById('docList');
  const empty=document.getElementById('docEmpty');
  if(!list) return;
  list.innerHTML='';
  if(complianceDocs.length===0){
    if(empty) empty.style.display='block';
    return;
  }
  if(empty) empty.style.display='none';
  complianceDocs.forEach(doc=>{
    const el=document.createElement('div');
    el.className='doc-item';
    const sizeMB=(doc.size/1024/1024).toFixed(2);
    el.innerHTML=`
      <div class="doc-icon">📄</div>
      <div class="doc-info">
        <div class="doc-name" title="${doc.name}">${doc.name}</div>
        <div class="doc-meta">${sizeMB} MB · ${doc.uploadedAt}</div>
        <div class="doc-meta" style="color:rgba(168,85,247,.7);margin-top:2px;font-size:8px">${doc.fileId}</div>
      </div>
      <button class="doc-del rpl" onclick="deleteDoc('${doc.fileId}')">${lang==='ar'?'حذف':'Delete'}</button>`;
    list.appendChild(el);
  });
}

  const rec=inspections.find(r=>r.id==id); const recLang=rec?.lang||'en';

function applyFilter(){
  document.querySelectorAll('.def-card').forEach(c=>{
    if(filter==='all'){ c.style.display=''; return; }
    if(filter==='resolved'){ c.style.display=c.classList.contains('done')?'':'none'; return; }

  });
}

function updateRiskCard(){
  if(!inspections.length) return;
  const avg=Math.round(inspections.reduce((a,r)=>a+(r.result.riskScore||0),0)/inspections.length);
  const clr=avg>70?'var(--red2)':avg>40?'var(--amber2)':'var(--green2)';
  document.getElementById('riskCard').classList.add('on');
  document.getElementById('riskNum').textContent=avg; document.getElementById('riskNum').style.color=clr;
  document.getElementById('riskFill').style.width=avg+'%'; document.getElementById('riskFill').style.background=clr;
  document.getElementById('riskLbl').textContent=avg>70?t('riskHigh'):avg>40?t('riskMed'):t('riskLow');
}

function updateDashboard(){

  document.getElementById('dTotal').textContent=inspections.length;
  document.getElementById('dIssues').textContent=all.length; document.getElementById('dFixed').textContent=res;
  document.getElementById('dCrit').textContent=crit+high; document.getElementById('dMinor').textContent=med+low;
  document.getElementById('dSC').textContent=crit; document.getElementById('dSH').textContent=high;
  document.getElementById('dSM').textContent=med; document.getElementById('dSL').textContent=low;
  const areas=inspections.reduce((acc,r)=>{ const c=(r.result.defects||[]).length; acc[r.area]=(acc[r.area]||0)+c; return acc; },{});
  const rows=document.getElementById('areaRows'),maxV=Math.max(...Object.values(areas),1);
